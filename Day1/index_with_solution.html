For Loops: Do Things Over and Over!

Try-except Blocks: Exception Handling!

Handle errors gracefully without crashing your program!


# Example of handling division by zero
def safe_divide(a, b):
    try: 
        result = a / b
        return result
    except ZeroDivisionError:
        return "Error: Division by zero is not allowed."

# Example usage
print(safe_divide(10, 2))
print(safe_divide(10, 0))

Finally Blocks: Run Code Regardless of Exceptions

Use finally to run code that should always execute, like cleanup actions.


try:
    file = open("example.txt", "r")
    content = file.read()
    print(content)
except FileNotFoundError:
    print("File not found.")
finally:
    print("Closing file...")
    file.close()

Boss Challenge: Rock, Paper, Scissors

Your task: Build the ultimate Rock, Paper, Scissors game!

Expected Input: User types "rock", "paper", or "scissors"

Expected Output: Game results and score updates

Example:
You chose: rock
Computer chose: scissors
You win this round!
Score - You: 1 | Computer: 0

🎮 Boss Challenge: Sample Solution


import random

def play_rps():
    choices = ["rock", "paper", "scissors"]
    player_score = 0
    computer_score = 0
    
    print("ROCK PAPER SCISSORS - ULTIMATE EDITION!")
    print("First to 3 wins!\n")
    
    while player_score < 3 and computer_score < 3:
        player = input("Choose (rock/paper/scissors): ").lower()
        computer = random.choice(choices)
        
        print(f"\nYou chose: {player}")
        print(f"Computer chose: {computer}")
        
        if player == computer:
            print("It's a tie!")
        elif (player == "rock" and computer == "scissors") or \
             (player == "paper" and computer == "rock") or \
             (player == "scissors" and computer == "paper"):
            print("You win this round!")
            player_score += 1
        else:
            print("Computer wins this round!")
            computer_score += 1
        
        print(f"Score - You: {player_score} | Computer: {computer_score}\n")
    
    if player_score > computer_score:
        print("YOU ARE THE CHAMPION!")
    else:
        print("Computer wins! Better luck next time!")

play_rps()

Hands-on Project: Movie Recommendation System

Let's build a simple movie recommendation system using Python!

Task 1: Find User Preferences

  • Given a list of movies user watched (3 movies) as well as a list of genres for each movie
  • Determine the most common genres as user preferences (if there are multiple genres, consider all of them)
  • Example: User watched "The Godfather", "Pulp Fiction", "Forrest Gump" → Preferred genre: Drama

Task 1: Sample Solution


def get_preferred_genres(movie_list):
    genre_counts = {}

    for movie in movie_list:
        genres = movie_genres.get(movie)
        if genres:
            for genre in genres:
                if genre in genre_counts:
                    genre_counts[genre] += 1
                else:
                    genre_counts[genre] = 1
        else:
            print(f"Warning: '{movie}' not found in the database.")

    if not genre_counts:
        return []  # No valid genres found

    # Find the max count
    max_count = max(genre_counts.values())

    # Get all genres with the max count
    preferred_genres = [genre for genre, count in genre_counts.items() if count == max_count]

    return preferred_genres

# Example usage:
user_movies = ["Snatch", "Lock, Stock and Two Smoking Barrels", "Reservoir Dogs"]
preferred_genres = get_preferred_genres(user_movies)
print(preferred_genres)

def recommend_movies(user_movies, preferred_genres):
    if len(preferred_genres) > 3:
        preferred_genres = random.sample(preferred_genres, 3)

    recommendations = []

    for movie, genres in movie_genres.items():
        if movie in user_movies:
            continue
        match_count = sum(1 for genre in genres if genre in preferred_genres)
        if match_count > 0:
            recommendations.append((match_count, movie))

    # Sort by number of matched genres (descending), then by name
    recommendations.sort(key=lambda x: (-x[0], x[1]))

    # Uncomment the following line to see what the sorted recommendations look like
    # print(recommendations)

    top_3 = []
    index = 0
    while len(top_3) < 3 and index < len(recommendations):
        top_3.append(recommendations[index][1])
        index += 1

    return top_3


# Example usage:
user_movies = ["Snatch", "Lock, Stock and Two Smoking Barrels", "Reservoir Dogs"]
preferred_genres = get_preferred_genres(user_movies)
recommended = recommend_movies(user_movies, preferred_genres)
print("---------------")
print("Preferred genres:", preferred_genres)
print("Recommended movies:", recommended)

Task 3: Adaptive Recommendations for Changing Preferences (Optional)

  • A user watches 100 Action movies, then his friend watches 3 Comedy movies.
    Using the original algorithm, recommendations still favor Action movies.
  • Goal: Recognize recent viewing trends over historical data, the preference from the original algorithm is Action, but now it's Comedy
  • Hint: Use a time decay factor to give more weight to recent views

Task 3: Sample Solution


def get_recent_weighted_genres(watched_movies):
    genre_scores = {}
    weights = [0.5, 0.3, 0.1, 0.1]  # latest 4 movie weights (in order: most recent first)

    # Look at the last 4 movies (if available)
    last_movies = watched_movies[-4:]
    last_movies.reverse()  # Make most recent first
    for i in range(len(last_movies)):
        movie = last_movies[i]
        weight = weights[i]
        genres = movie_genres.get(movie, [])
        for genre in genres:
            if genre in genre_scores:
                genre_scores[genre] += weight
            else:
                genre_scores[genre] = weight

    # Get top genre(s)
    max_score = max(genre_scores.values())
    preferred = [genre for genre, score in genre_scores.items() if score == max_score]
    return preferred

# Simulate a long history of action movies, but recent 4 are different
watched_movies = ["Spider-Man: Into the Spider-Verse"] * 100 + [
    "Parasite", "City Lights", "Amelie", "The Grand Budapest Hotel"  # Most recent 4, The Grand Budapest Hotel is the most recent one
]

preferred = get_recent_weighted_genres(watched_movies)
recommended = recommend_movies(watched_movies, preferred)

print("Top preferred genres (weighted recent 4):", preferred)
print("Recommended movies:", recommended)
